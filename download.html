Hazelcast IMDG can also be retrieved using the following tools:

Maven
Graddle
Docker
Helm
<dependencies>
    <dependency>
        <groupId>com.hazelcast</groupId>
        <artifactId>hazelcast-all</artifactId>
        <version>4.0.1</version>
    </dependency>
</dependencies>
compile 'com.hazelcast:hazelcast-all:4.0.1'
docker run hazelcast/hazelcast:v4.0.1
helm repo add hazelcast https://hazelcast.github.io/charts/
helm repo update

helm install --set image.tag=4.0.1 hazelcast/hazelcast
